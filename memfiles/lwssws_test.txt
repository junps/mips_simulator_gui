    .data
init.1:
    0x40000000
init.2:
    0x40400000

    .globl start
print_word:
    out $a0
    srl $a0, $a0, 8
    out $a0
    srl $a0, $a0, 8
    out $a0
    srl $a0, $a0, 8
    out $a0
    jr $ra
start:
    in $v0
    out $v0
    lw.s $f0, init.1
    lw.s $f1, init.2
    add.s $f2, $f0, $f1
    sw.s $f2, 0($sp)
    lw $v0, 0($sp)
    move $a0, $v0
    jal print_word
